// ================================================================
//  Ghost Killer — Server-Authoritative Game State
// ================================================================

const TILE_SIZE = 48;
const MAP_W = 60;
const MAP_H = 60;
const NUM_ROOMS = 18;
const ENTITY_RADIUS = 14;
const GHOST_RADIUS = 16;

const DOOR_HPS = [100, 300, 800, 2000, 5000, 12000, 25000, 50000, 100000, 250000, 500000, 1000000];
const BED_INCOME = [1, 2, 4, 8, 16, 32, 64, 128, 256, 512, 1024, 2048];
const TURRET_DMG = [10, 25, 60, 150, 400, 1000, 2500, 5000, 12000, 25000, 60000, 150000];
const TURRET_RNG = [250, 275, 300, 325, 350, 375, 400, 425, 450, 475, 500, 550];
const GEN_INCOME = [1, 2, 4, 8, 16, 32, 64, 128, 256, 512, 1024, 2048];
const MINE_INCOME = {
    copper: [2, 4, 8, 16, 32, 64],
    silver: [6, 12, 24, 48, 96, 192],
    gold: [15, 30, 60, 120, 240, 480],
};
const PLAYER_COLORS = ['#38bdf8', '#a3e635', '#f472b6', '#fbbf24', '#c084fc', '#2dd4bf', '#fb923c'];

const ghostSpots = [
    { x: Math.floor(MAP_W / 2), y: 2 },
    { x: Math.floor(MAP_W / 2), y: MAP_H - 3 },
    { x: 2, y: Math.floor(MAP_H / 2) },
    { x: MAP_W - 3, y: Math.floor(MAP_H / 2) },
];

function dist(x1, y1, x2, y2) { return Math.sqrt((x2 - x1) ** 2 + (y2 - y1) ** 2); }

function generateMap() {
    const map = [];
    for (let y = 0; y < MAP_H; y++) { map[y] = []; for (let x = 0; x < MAP_W; x++) map[y][x] = 0; }
    for (let y = 0; y < MAP_H; y++) for (let x = 0; x < MAP_W; x++)
        if (x < 4 || x >= MAP_W - 4 || y < 4 || y >= MAP_H - 4) map[y][x] = 1;
    const cx = Math.floor(MAP_W / 2), cy = Math.floor(MAP_H / 2);
    for (let y = 1; y < 4; y++) for (let x = cx - 1; x <= cx + 1; x++) map[y][x] = 0;
    for (let y = MAP_H - 4; y < MAP_H - 1; y++) for (let x = cx - 1; x <= cx + 1; x++) map[y][x] = 0;
    for (let x = 1; x < 4; x++) for (let y = cy - 1; y <= cy + 1; y++) map[y][x] = 0;
    for (let x = MAP_W - 4; x < MAP_W - 1; x++) for (let y = cy - 1; y <= cy + 1; y++) map[y][x] = 0;
    return map;
}

function generateRooms(map) {
    const rooms = [];
    const cx = Math.floor(MAP_W / 2), cy = Math.floor(MAP_H / 2);
    let attempts = 0;
    while (rooms.length < NUM_ROOMS && attempts < 2000) {
        attempts++;
        const rw = Math.floor(Math.random() * 5) + 5;
        const rh = Math.floor(Math.random() * 5) + 5;
        const rx = Math.floor(Math.random() * (MAP_W - rw - 12)) + 6;
        const ry = Math.floor(Math.random() * (MAP_H - rh - 12)) + 6;
        let overlap = false;
        if (rx < cx + 7 && rx + rw > cx - 7 && ry < cy + 7 && ry + rh > cy - 7) overlap = true;
        if (!overlap) for (const r of rooms)
            if (rx < r.x + r.w + 3 && rx + rw + 3 > r.x && ry < r.y + r.h + 3 && ry + rh + 3 > r.y) { overlap = true; break; }
        if (overlap) continue;
        for (let y = ry - 1; y <= ry + rh; y++) for (let x = rx - 1; x <= rx + rw; x++)
            if (x === rx - 1 || x === rx + rw || y === ry - 1 || y === ry + rh) map[y][x] = 1;
        const doorSide = Math.floor(Math.random() * 4);
        let ddx, ddy;
        if (doorSide === 0) { ddx = rx + Math.floor(rw / 2); ddy = ry - 1; }
        else if (doorSide === 1) { ddx = rx + Math.floor(rw / 2); ddy = ry + rh; }
        else if (doorSide === 2) { ddx = rx - 1; ddy = ry + Math.floor(rh / 2); }
        else { ddx = rx + rw; ddy = ry + Math.floor(rh / 2); }
        map[ddy][ddx] = 2;
        let bx = rx + Math.floor(rw / 2), by = ry + Math.floor(rh / 2);
        if (doorSide === 0) by = ry + rh - 2;
        else if (doorSide === 1) by = ry + 1;
        else if (doorSide === 2) bx = rx + rw - 2;
        else bx = rx + 1;
        rooms.push({
            id: rooms.length, x: rx, y: ry, w: rw, h: rh,
            door: { x: ddx, y: ddy, hp: DOOR_HPS[0], maxHp: DOOR_HPS[0], level: 0, hitTimer: 0, shieldActive: false, shieldUsed: false },
            bed: { x: bx, y: by, level: 0 },
            turrets: [], generators: [], mines: [], owner: null,
            _hasGrenadeLauncher: false, _hasGhostNet: false, _hasShield: false, _hasRepairTool: false,
        });
    }
    return rooms;
}

// ── COLLISION ─────────────────────────────────────────────────────
// tile=1: solid wall — always blocks.
// tile=2: door — blocks while hp > 0 regardless of ownership.
function tileBlocks(map, rooms, tx, ty) {
    if (tx < 0 || tx >= MAP_W || ty < 0 || ty >= MAP_H) return true;
    if (map[ty][tx] === 1) return true;
    if (map[ty][tx] === 2) {
        const room = rooms.find(r => r.door.x === tx && r.door.y === ty);
        if (!room) return false;
        return room.door.hp > 0;  // blocks while any hp remains, owned or not
    }
    return false;
}

function checkWallCollision(map, rooms, nx, ny, radius) {
    const r = radius - 2;
    const pts = [
        { x: nx - r, y: ny - r }, { x: nx, y: ny - r }, { x: nx + r, y: ny - r },
        { x: nx - r, y: ny }, { x: nx + r, y: ny },
        { x: nx - r, y: ny + r }, { x: nx, y: ny + r }, { x: nx + r, y: ny + r },
    ];
    for (const p of pts)
        if (tileBlocks(map, rooms, Math.floor(p.x / TILE_SIZE), Math.floor(p.y / TILE_SIZE))) return true;
    return false;
}

function moveWithCollision(map, rooms, x, y, vx, vy, speed, dt, radius) {
    const STEPS = 6;
    let nx = x, ny = y;
    const dx = (vx * speed * dt) / STEPS;
    const dy = (vy * speed * dt) / STEPS;
    for (let i = 0; i < STEPS; i++) {
        if (!checkWallCollision(map, rooms, nx + dx, ny, radius)) nx += dx;
        if (!checkWallCollision(map, rooms, nx, ny + dy, radius)) ny += dy;
    }
    return { x: nx, y: ny };
}

function validateClientPos(map, rooms, sx, sy, cx, cy, radius) {
    if (!checkWallCollision(map, rooms, cx, cy, radius)) return { x: cx, y: cy };
    return { x: sx, y: sy };
}

// ── BFS ───────────────────────────────────────────────────────────
// Destination tile always walkable (so ghost can path TO the door).
// All other door tiles are treated as walls.
function getPathNextNode(map, rooms, sx, sy, tx, ty) {
    if (sx === tx && sy === ty) return null;
    const visited = new Uint8Array(MAP_W * MAP_H);
    const queue = [{ x: sx, y: sy, path: [] }];
    visited[sy * MAP_W + sx] = 1;
    const dirs = [[0, -1], [0, 1], [-1, 0], [1, 0]];
    let steps = 0;
    while (queue.length > 0 && steps++ < 5000) {
        const curr = queue.shift();
        if (curr.x === tx && curr.y === ty) return curr.path[0] || null;
        for (const [ddx, ddy] of dirs) {
            const nx = curr.x + ddx, ny = curr.y + ddy;
            if (nx < 0 || nx >= MAP_W || ny < 0 || ny >= MAP_H || visited[ny * MAP_W + nx]) continue;
            visited[ny * MAP_W + nx] = 1;
            const isDest = (nx === tx && ny === ty);
            let walkable = true;
            if (map[ny][nx] === 1) walkable = false;
            else if (map[ny][nx] === 2 && !isDest) walkable = false;  // all doors block BFS
            if (walkable) queue.push({ x: nx, y: ny, path: curr.path.concat({ x: nx, y: ny }) });
        }
    }
    return null;
}

// ── SERVER GAME ───────────────────────────────────────────────────
class ServerGame {
    constructor(roomId, playerList) {
        this.roomId = roomId;
        this.map = generateMap();
        this.rooms = generateRooms(this.map);
        this.players = {};
        this.ghost = null;
        this.ghostIsBot = false;
        this.ghostTimer = 25;
        this.gameOver = false;
        this.winner = null;
        this.startTime = Date.now();

        const ids = playerList.map(p => p.id);
        const ghostId = ids[Math.floor(Math.random() * ids.length)];
        const cx = Math.floor(MAP_W / 2) * TILE_SIZE + TILE_SIZE / 2;
        const cy = Math.floor(MAP_H / 2) * TILE_SIZE + TILE_SIZE / 2;
        const sp = ghostSpots[Math.floor(Math.random() * ghostSpots.length)];

        playerList.forEach((p, i) => {
            const isGhost = p.id === ghostId;
            this.players[p.id] = {
                id: p.id, name: p.name,
                color: isGhost ? '#ef4444' : PLAYER_COLORS[i % PLAYER_COLORS.length],
                isGhost,
                x: isGhost ? sp.x * TILE_SIZE + TILE_SIZE / 2 : cx + (Math.random() - .5) * 100,
                y: isGhost ? sp.y * TILE_SIZE + TILE_SIZE / 2 : cy + (Math.random() - .5) * 100,
                vx: 0, vy: 0, facing: 0,
                speed: isGhost ? 130 : 220,
                state: 'ROAMING',
                hp: isGhost ? 1000 : 100, maxHp: isGhost ? 1000 : 100,
                coins: isGhost ? 0 : 50, energy: 0, room: null,
                ghostLevel: 1, ghostXp: 0,
                ghostAttackTimer: 0,  // for AI door attacks
                ghostDoorTimer: 0,    // FIX: separate timer for human door attacks
                ghostBodyTimer: 0,    // FIX: separate timer for human body attacks
                ghostState: 'WAITING', ghostTarget: null,
                ghostPathTimer: 0, _ghostPathCache: null, ghostStunTimer: 0,
                repairActive: false, repairActiveTimer: 0, repairCooldown: 0,
                _resourceTimer: 0,
            };
        });
        this.ghost = Object.values(this.players).find(p => p.isGhost);
    }

    tick(dt) {
        if (this.gameOver) return null;

        if (this.ghostTimer > 0) {
            this.ghostTimer = Math.max(0, this.ghostTimer - dt);
            if (this.ghostTimer <= 0 && this.ghost) {
                this.ghostTimer = 0;
                this.ghost.ghostState = 'HUNTING';
            }
        }

        for (const p of Object.values(this.players)) {
            if (p.isGhost || p.state === 'DEAD') continue;
            if (p.state === 'SLEEPING' && p.room !== null) {
                const room = this.rooms[p.room];
                if (!room) continue;
                p._resourceTimer += dt;
                if (p._resourceTimer >= 1.0) {
                    p._resourceTimer -= 1.0;
                    let cInc = BED_INCOME[room.bed.level];
                    for (const g of room.generators) cInc += GEN_INCOME[g.level - 1];
                    for (const m of room.mines) { const arr = MINE_INCOME[m.type]; cInc += arr[Math.min(m.level - 1, arr.length - 1)]; }
                    p.coins += cInc;
                    p.energy += room.generators.reduce((s, g) => s + GEN_INCOME[g.level - 1], 0);
                }
                if (p.repairActive) {
                    p.repairActiveTimer -= dt;
                    const rate = (room._hasRepairTool ? 0.15 : 0.08) * room.door.maxHp * dt;
                    room.door.hp = Math.min(room.door.maxHp, room.door.hp + rate);
                    if (p.repairActiveTimer <= 0) { p.repairActive = false; p.repairCooldown = 60; }
                }
                if (p.repairCooldown > 0) p.repairCooldown = Math.max(0, p.repairCooldown - dt);
            }
        }

        if (this.ghost && this.ghost.state !== 'DEAD') {
            if (this.ghostTimer <= 0) {
                if (this.ghostIsBot) this._updateGhost(dt);
                else this._applyGhostPlayerInput(dt);
            }
        }

        this._updateTurrets(dt);
        for (const r of this.rooms) if (r.door.hitTimer > 0) r.door.hitTimer -= dt;

        const aliveHumans = Object.values(this.players).filter(p => !p.isGhost && p.state !== 'DEAD');
        if (aliveHumans.length === 0 && Object.keys(this.players).length > 1) { this.gameOver = true; this.winner = 'ghost'; }
        if (this.ghost && this.ghost.state === 'DEAD') { this.gameOver = true; this.winner = 'survivors'; }

        return this.getState();
    }

    _updateGhost(dt) {
        const g = this.ghost;
        if (!g || g.state === 'DEAD') return;
        if (g.ghostStunTimer > 0) { g.ghostStunTimer -= dt; return; }

        const xpReq = g.ghostLevel * 200;
        g.ghostXp += 5 * dt;
        if (g.ghostXp >= xpReq) { g.ghostXp -= xpReq; g.ghostLevel++; g.maxHp = Math.floor(g.maxHp * 1.6); g.hp = g.maxHp; g.speed = Math.min(200, g.speed + 5); }
        if (g.ghostLevel >= 3 && g.hp < g.maxHp) g.hp = Math.min(g.maxHp, g.hp + g.ghostLevel * 2 * dt);

        const targets = Object.values(this.players).filter(p => !p.isGhost && p.state !== 'DEAD');
        if (!targets.length) return;
        const curTarget = g.ghostTarget && this.players[g.ghostTarget];
        if (!curTarget || curTarget.state === 'DEAD') {
            g.ghostTarget = targets.sort((a, b) => dist(g.x, g.y, a.x, a.y) - dist(g.x, g.y, b.x, b.y))[0].id;
            g.ghostState = 'HUNTING'; g._ghostPathCache = null;
        }

        const tgt = this.players[g.ghostTarget];
        if (!tgt || tgt.state === 'DEAD') { g.ghostTarget = null; g.ghostState = 'HUNTING'; return; }

        // ── STATE MACHINE — strict else-if so only ONE block fires per tick ──
        if (g.ghostState === 'HUNTING') {
            if (tgt.state === 'ROAMING') {
                this._bfsMove(g, Math.floor(tgt.x / TILE_SIZE), Math.floor(tgt.y / TILE_SIZE), tgt.x, tgt.y, dt);
                if (dist(g.x, g.y, tgt.x, tgt.y) < GHOST_RADIUS + ENTITY_RADIUS) {
                    tgt.state = 'DEAD'; tgt.hp = 0; g.ghostXp += 150;
                    g.ghostTarget = null; g._ghostPathCache = null;
                }
            } else if (tgt.room !== null) {
                const room = this.rooms[tgt.room];
                if (!room) { g.ghostTarget = null; return; }
                const ap = this._approachTile(room);
                this._bfsMove(g, ap.tx, ap.ty, ap.px, ap.py, dt);
                if (dist(g.x, g.y, ap.px, ap.py) < TILE_SIZE * 1.5) {
                    g.ghostState = 'APPROACHING_DOOR'; g._ghostPathCache = null;
                }
            }
        } else if (g.ghostState === 'APPROACHING_DOOR') {
            if (tgt.room === null) { g.ghostState = 'HUNTING'; return; }
            const room = this.rooms[tgt.room];
            if (!room) { g.ghostState = 'HUNTING'; return; }
            const ap = this._approachTile(room);
            const dd = dist(g.x, g.y, ap.px, ap.py);
            if (dd > 6) {
                const a = Math.atan2(ap.py - g.y, ap.px - g.x);
                const moved = moveWithCollision(this.map, this.rooms, g.x, g.y, Math.cos(a), Math.sin(a), g.speed * 0.6, dt, GHOST_RADIUS);
                g.x = moved.x; g.y = moved.y; g.facing = a;
            } else { g.ghostState = 'ATTACKING'; g.ghostAttackTimer = 0; }
        } else if (g.ghostState === 'ATTACKING') {
            if (tgt.room === null || tgt.state === 'DEAD') { g.ghostState = 'HUNTING'; return; }
            const room = this.rooms[tgt.room];
            if (!room) { g.ghostState = 'HUNTING'; return; }
            const door = room.door;
            if (door.hp <= 0) { g.ghostState = 'CROSSING'; g._ghostPathCache = null; return; }
            g.ghostAttackTimer += dt;
            if (g.ghostAttackTimer >= 0.4) {
                g.ghostAttackTimer = 0;
                const dmg = 30 + g.ghostLevel * 15;
                door.hp -= dmg; door.hitTimer = 0.2; g.ghostXp += 30;
                if (!door.shieldUsed && door.hp / door.maxHp < 0.3 && room._hasShield) {
                    door.shieldActive = true; door.shieldUsed = true;
                    door.hp = Math.min(door.hp + door.maxHp * 0.3, door.maxHp * 0.6);
                    room._hasShield = false;
                }
                if (door.hp <= 0) { door.hp = 0; g.ghostXp += 200; g.ghostState = 'CROSSING'; g._ghostPathCache = null; }
            }
            // Keep ghost glued at approach tile while hammering door
            const ap = this._approachTile(room);
            if (dist(g.x, g.y, ap.px, ap.py) > 8) {
                const a = Math.atan2(ap.py - g.y, ap.px - g.x);
                const glue = moveWithCollision(this.map, this.rooms, g.x, g.y, Math.cos(a), Math.sin(a), g.speed * 0.2, dt, GHOST_RADIUS);
                g.x = glue.x; g.y = glue.y;
            }
        } else if (g.ghostState === 'CROSSING') {
            if (tgt.state === 'DEAD' || tgt.room === null) { g.ghostState = 'HUNTING'; return; }
            const room = this.rooms[tgt.room];
            if (!room) { g.ghostState = 'HUNTING'; return; }
            if (room.door.hp > 0) { g.ghostState = 'ATTACKING'; g.ghostAttackTimer = 0; return; }
            const gpx = room.door.x * TILE_SIZE + TILE_SIZE / 2;
            const gpy = room.door.y * TILE_SIZE + TILE_SIZE / 2;
            const dd = dist(g.x, g.y, gpx, gpy);
            if (dd > 8) {
                const a = Math.atan2(gpy - g.y, gpx - g.x);
                const moved = moveWithCollision(this.map, this.rooms, g.x, g.y, Math.cos(a), Math.sin(a), g.speed * 1.4, dt, GHOST_RADIUS);
                g.x = moved.x; g.y = moved.y; g.facing = a;
            } else { g.ghostState = 'INSIDE'; g.ghostAttackTimer = 0; }
        } else if (g.ghostState === 'INSIDE') {
            if (tgt.state === 'DEAD' || tgt.room === null) {
                g.ghostState = 'HUNTING'; g.ghostTarget = null; g._ghostPathCache = null; return;
            }
            const dx = tgt.x - g.x, dy = tgt.y - g.y, dd = Math.sqrt(dx * dx + dy * dy);
            if (dd > 2) {
                const moved = moveWithCollision(this.map, this.rooms, g.x, g.y, dx / dd, dy / dd, g.speed, dt, GHOST_RADIUS);
                g.x = moved.x; g.y = moved.y; g.facing = Math.atan2(dy, dx);
            }
            if (dd < GHOST_RADIUS + ENTITY_RADIUS + 6) {
                g.ghostAttackTimer += dt;
                if (g.ghostAttackTimer >= 0.4) {
                    g.ghostAttackTimer = 0;
                    const dmg = 30 + g.ghostLevel * 15;
                    tgt.hp -= dmg;
                    if (tgt.hp <= 0) {
                        tgt.hp = 0; tgt.state = 'DEAD'; g.ghostXp += 200;
                        g.ghostTarget = null; g.ghostState = 'HUNTING'; g._ghostPathCache = null;
                    }
                }
            }
        }
    }

    // ── HUMAN GHOST INPUT ─────────────────────────────────────────
    // FIX: separate ghostDoorTimer and ghostBodyTimer so door attacks
    // and player attacks don't share/corrupt the same counter.
    _applyGhostPlayerInput(dt) {
        const g = this.ghost;
        if (!g || g.state === 'DEAD' || this.ghostTimer > 0) return;

        if (g.ghostStunTimer > 0) { g.ghostStunTimer -= dt; return; }

        // Move
        const moved = moveWithCollision(this.map, this.rooms, g.x, g.y, g.vx || 0, g.vy || 0, g.speed, dt, GHOST_RADIUS);
        g.x = moved.x; g.y = moved.y;

        // Tick both attack timers every frame
        g.ghostDoorTimer = (g.ghostDoorTimer || 0) + dt;
        g.ghostBodyTimer = (g.ghostBodyTimer || 0) + dt;

        for (const p of Object.values(this.players)) {
            if (p.isGhost || p.state === 'DEAD') continue;
            const d = dist(g.x, g.y, p.x, p.y);

            if (p.state === 'ROAMING') {
                if (d < GHOST_RADIUS + ENTITY_RADIUS) {
                    p.state = 'DEAD'; p.hp = 0; g.ghostXp += 200;
                }
                continue;
            }

            if (p.state === 'SLEEPING' && p.room !== null) {
                const room = this.rooms[p.room];
                if (!room) continue;
                const door = room.door;
                const doorPx = door.x * TILE_SIZE + TILE_SIZE / 2;
                const doorPy = door.y * TILE_SIZE + TILE_SIZE / 2;
                const doorDist = dist(g.x, g.y, doorPx, doorPy);

                // Attack door (separate timer — ghostDoorTimer)
                if (door.hp > 0 && doorDist < TILE_SIZE * 1.8) {
                    if (g.ghostDoorTimer >= 0.4) {
                        g.ghostDoorTimer = 0;
                        const dmg = 30 + g.ghostLevel * 15;
                        door.hp -= dmg; door.hitTimer = 0.2; g.ghostXp += 30;
                        if (!door.shieldUsed && door.hp / door.maxHp < 0.3 && room._hasShield) {
                            door.shieldActive = true; door.shieldUsed = true;
                            door.hp = Math.min(door.hp + door.maxHp * 0.3, door.maxHp * 0.6);
                            room._hasShield = false;
                        }
                        if (door.hp <= 0) { door.hp = 0; g.ghostXp += 200; }
                    }
                }

                // Attack survivor body (separate timer — ghostBodyTimer)
                if (door.hp <= 0 && d < GHOST_RADIUS + ENTITY_RADIUS + 6) {
                    if (g.ghostBodyTimer >= 0.4) {
                        g.ghostBodyTimer = 0;
                        const dmg = 30 + g.ghostLevel * 15;
                        p.hp -= dmg;
                        if (p.hp <= 0) { p.hp = 0; p.state = 'DEAD'; g.ghostXp += 200; }
                    }
                }
            }
        }

        // XP / level-up
        const xpReq = g.ghostLevel * 200;
        if (g.ghostXp >= xpReq) {
            g.ghostXp -= xpReq; g.ghostLevel++;
            g.maxHp = Math.floor(g.maxHp * 1.6); g.hp = g.maxHp;
            g.speed = Math.min(200, g.speed + 5);
        }
        if (g.ghostLevel >= 3 && g.hp < g.maxHp)
            g.hp = Math.min(g.maxHp, g.hp + g.ghostLevel * 2 * dt);
    }

    _bfsMove(g, destTx, destTy, destPx, destPy, dt) {
        const gTx = Math.floor(g.x / TILE_SIZE), gTy = Math.floor(g.y / TILE_SIZE);
        g.ghostPathTimer = (g.ghostPathTimer || 0) + dt;
        if (!g._ghostPathCache || g.ghostPathTimer > 0.4) {
            g.ghostPathTimer = 0;
            g._ghostPathCache = getPathNextNode(this.map, this.rooms, gTx, gTy, destTx, destTy);
        }
        let tpx = destPx, tpy = destPy;
        if (g._ghostPathCache) {
            tpx = g._ghostPathCache.x * TILE_SIZE + TILE_SIZE / 2;
            tpy = g._ghostPathCache.y * TILE_SIZE + TILE_SIZE / 2;
            if (dist(g.x, g.y, tpx, tpy) < TILE_SIZE * 0.6) g._ghostPathCache = null;
        }
        const a = Math.atan2(tpy - g.y, tpx - g.x);
        const moved = moveWithCollision(this.map, this.rooms, g.x, g.y, Math.cos(a), Math.sin(a), g.speed, dt, GHOST_RADIUS);
        g.x = moved.x; g.y = moved.y; g.facing = a;
    }

    _approachTile(room) {
        const { x: gx, y: gy } = room.door, { x: rx, y: ry, w, h } = room;
        let tx, ty;
        if (gy === ry - 1) { tx = gx; ty = gy - 1; }
        else if (gy === ry + h) { tx = gx; ty = gy + 1; }
        else if (gx === rx - 1) { tx = gx - 1; ty = gy; }
        else { tx = gx + 1; ty = gy; }
        return { tx, ty, px: tx * TILE_SIZE + TILE_SIZE / 2, py: ty * TILE_SIZE + TILE_SIZE / 2 };
    }

    _updateTurrets(dt) {
        const g = this.ghost;
        if (!g || g.state === 'DEAD' || g.ghostState === 'WAITING') return;
        for (const room of this.rooms) {
            if (!room.owner) continue;
            const owner = this.players[room.owner];
            if (!owner || owner.state === 'DEAD') continue;
            for (const t of room.turrets) {
                const ttx = t.x * TILE_SIZE + TILE_SIZE / 2, tty = t.y * TILE_SIZE + TILE_SIZE / 2;
                const range = TURRET_RNG[t.level - 1];
                const dg = dist(ttx, tty, g.x, g.y);
                if (dg < range) t.angle = Math.atan2(g.y - tty, g.x - ttx);
                t.lastFire = (t.lastFire || 0) + dt;
                if (t.lastFire >= 0.8 && dg < range) {
                    t.lastFire = 0;
                    g.hp -= TURRET_DMG[t.level - 1];
                    if (g.hp <= 0) { g.hp = 0; g.state = 'DEAD'; this.gameOver = true; this.winner = 'survivors'; }
                }
            }
        }
    }

    playerSleep(socketId, roomIdx) {
        const p = this.players[socketId], room = this.rooms[roomIdx];
        if (!p || !room || room.owner || p.state !== 'ROAMING') return false;
        p.state = 'SLEEPING'; p.room = roomIdx;
        p.x = room.bed.x * TILE_SIZE + TILE_SIZE / 2;
        p.y = room.bed.y * TILE_SIZE + TILE_SIZE / 2;
        room.owner = socketId; return true;
    }

    playerBuild(socketId, tileX, tileY, buildType) {
        const p = this.players[socketId];
        if (!p || p.state !== 'SLEEPING' || p.room === null) return { ok: false };
        const room = this.rooms[p.room];
        if (!room) return { ok: false };
        if (tileX < room.x || tileX >= room.x + room.w || tileY < room.y || tileY >= room.y + room.h) return { ok: false };
        const COSTS = { turret: 30, generator: 50, copper_mine: 40, silver_mine: 120, gold_mine: 300, grenade_launcher: 200, ghost_net: 350, door_shield: 250, repair_tool: 150 };
        const cost = COSTS[buildType];
        if (!cost || p.coins < cost) return { ok: false, reason: 'Not enough coins' };
        const occupied = [...room.turrets, ...room.generators, ...room.mines].some(t => t.x === tileX && t.y === tileY);
        if (occupied || (tileX === room.bed.x && tileY === room.bed.y)) return { ok: false, reason: 'Tile occupied' };
        p.coins -= cost;
        if (buildType === 'turret') room.turrets.push({ x: tileX, y: tileY, level: 1, lastFire: 0, angle: 0 });
        else if (buildType === 'generator') room.generators.push({ x: tileX, y: tileY, level: 1 });
        else if (['copper_mine', 'silver_mine', 'gold_mine'].includes(buildType)) room.mines.push({ x: tileX, y: tileY, type: buildType.replace('_mine', ''), level: 1 });
        else if (buildType === 'grenade_launcher') room._hasGrenadeLauncher = true;
        else if (buildType === 'ghost_net') room._hasGhostNet = true;
        else if (buildType === 'door_shield') room._hasShield = true;
        else if (buildType === 'repair_tool') room._hasRepairTool = true;
        return { ok: true };
    }

    playerUpgrade(socketId, type, tileX, tileY) {
        const p = this.players[socketId];
        if (!p || p.state !== 'SLEEPING' || p.room === null) return { ok: false };
        const room = this.rooms[p.room];
        if (!room) return { ok: false };
        const BC = [10, 25, 60, 150, 300, 600, 1500, 3000, 6000, 12000, 25000, 50000];
        const DC = [20, 50, 120, 250, 500, 1000, 2500, 5000, 10000, 20000, 40000, 80000];
        const TC = [30, 80, 200, 450, 1000, 2500, 6000, 12000, 25000, 50000, 100000, 250000];
        const GC = [50, 120, 300, 700, 1500, 3000, 7000, 15000, 30000, 60000, 120000, 300000];
        const MC = { copper: [40, 100, 250, 600, 1200, 2500], silver: [120, 300, 750, 1800, 3500, 7000], gold: [300, 750, 1800, 4000, 8000, 16000] };
        let cost = 0, obj = null;
        if (type === 'bed') { obj = room.bed; cost = BC[obj.level]; }
        else if (type === 'door') { obj = room.door; cost = DC[obj.level]; }
        else if (type === 'turret') { obj = room.turrets.find(t => t.x === tileX && t.y === tileY); if (obj) cost = TC[obj.level]; }
        else if (type === 'generator') { obj = room.generators.find(t => t.x === tileX && t.y === tileY); if (obj) cost = GC[obj.level]; }
        else if (type === 'mine') { obj = room.mines.find(t => t.x === tileX && t.y === tileY); if (obj) cost = (MC[obj.type] || [])[obj.level] || 0; }
        if (!obj || p.coins < cost || cost === 0) return { ok: false };
        p.coins -= cost; obj.level++;
        if (type === 'door') { obj.maxHp = DOOR_HPS[obj.level]; obj.hp = obj.maxHp; }
        return { ok: true, updatedObj: JSON.parse(JSON.stringify(obj)) };
    }

    playerRepair(socketId) {
        const p = this.players[socketId];
        if (!p || p.state !== 'SLEEPING' || p.room === null) return false;
        if (p.repairCooldown > 0 || p.repairActive) return false;
        p.repairActive = true; p.repairActiveTimer = 5; return true;
    }

    playerUseGrenade(socketId) {
        const p = this.players[socketId];
        if (!p || p.state !== 'SLEEPING' || p.room === null) return false;
        const room = this.rooms[p.room];
        if (!room._hasGrenadeLauncher || !this.ghost || this.ghost.state === 'DEAD') return false;
        room._hasGrenadeLauncher = false;
        const dmg = Math.floor(this.ghost.maxHp * 0.4 + 400);
        this.ghost.hp -= dmg;
        if (this.ghost.hp <= 0) { this.ghost.hp = 0; this.ghost.state = 'DEAD'; this.gameOver = true; this.winner = 'survivors'; }
        return { ok: true, dmg };
    }

    playerUseGhostNet(socketId) {
        const p = this.players[socketId];
        if (!p || p.state !== 'SLEEPING' || p.room === null) return false;
        const room = this.rooms[p.room];
        if (!room._hasGhostNet || !this.ghost) return false;
        room._hasGhostNet = false; this.ghost.ghostStunTimer = 3; return true;
    }

    playerMove(socketId, x, y, vx, vy, facing) {
        const p = this.players[socketId];
        if (!p || p.isGhost || p.state !== 'ROAMING') return;
        const v = validateClientPos(this.map, this.rooms, p.x, p.y, x, y, ENTITY_RADIUS);
        p.x = v.x; p.y = v.y; p.vx = vx; p.vy = vy; p.facing = facing;
    }

    ghostMove(socketId, vx, vy, facing) {
        const p = this.players[socketId];
        if (!p || !p.isGhost || p.state === 'DEAD') return;
        if (this.ghostTimer > 0) return;
        this.ghostIsBot = false;
        const mag = Math.sqrt(vx * vx + vy * vy);
        if (mag > 1) { vx /= mag; vy /= mag; }
        p.vx = vx; p.vy = vy;
        if (vx !== 0 || vy !== 0) p.facing = facing;
    }

    ghostDisconnected() {
        this.ghostIsBot = true;
        if (this.ghost) this.ghost.ghostState = 'HUNTING';
    }

    getState() {
        return { players: this.players, rooms: this.rooms, map: null, ghostTimer: this.ghostTimer, gameOver: this.gameOver, winner: this.winner };
    }
    getInitialState() {
        return { players: this.players, rooms: this.rooms, map: this.map, ghostTimer: this.ghostTimer };
    }
    getDuration() { return Math.floor((Date.now() - this.startTime) / 1000); }
}

        // ── CLIENT-SIDE COLLISION ────────────────────────────────────────
        // Mirrors the server logic so movement feels responsive.
        // Server is authoritative and will correct any discrepancy.
        function clientTileBlocks(tx, ty, isGhostMover) {
            if (!map || tx < 0 || tx >= MAP_W || ty < 0 || ty >= MAP_H) return true;
            if (map[ty][tx] === 1) return true;
            if (map[ty][tx] === 2) {
                const room = rooms && rooms.find(r => r.door.x === tx && r.door.y === ty);
                if (!room) return false;
                if (room.door.hp <= 0) return false; // broken door — always passable
                if (isGhostMover) return true;     // ghost blocked by ALL intact doors
                return !!room.owner;              // survivor blocked only by OWNED intact doors
            }
            return false;
        }

        // 8-point perimeter check, radius slightly inset so entity doesn't stick
        function clientWallHit(nx, ny, radius, isGhostMover) {
            const r = radius - 2;
            const pts = [
                { x: nx - r, y: ny - r }, { x: nx, y: ny - r }, { x: nx + r, y: ny - r },
                { x: nx - r, y: ny }, { x: nx + r, y: ny },
                { x: nx - r, y: ny + r }, { x: nx, y: ny + r }, { x: nx + r, y: ny + r },
            ];
            for (const p of pts) {
                if (clientTileBlocks(Math.floor(p.x / TILE_SIZE), Math.floor(p.y / TILE_SIZE), isGhostMover)) return true;
            }
            return false;
        }

        // Wall-sliding movement — returns new {x,y}, never teleports through walls
        function clientMoveCollide(x, y, vx, vy, speed, dt, radius, isGhostMover) {
            const steps = 4;
            let nx = x, ny = y;
            const dx = (vx * speed * dt) / steps;
            const dy = (vy * speed * dt) / steps;
            for (let i = 0; i < steps; i++) {
                if (!clientWallHit(nx + dx, ny, radius, isGhostMover)) nx += dx;
                if (!clientWallHit(nx, ny + dy, radius, isGhostMover)) ny += dy;
            }
            return { x: nx, y: ny };
        }

            if (!localInitDone) { localX = me.x; localY = me.y; localInitDone = true; }

            let vx = 0, vy = 0;
            if (keys.w || keys.ArrowUp) vy = -1;
            if (keys.s || keys.ArrowDown) vy = 1;
            if (keys.a || keys.ArrowLeft) vx = -1;
            if (keys.d || keys.ArrowRight) vx = 1;
            if (joyActive) { vx = joyDelta.x; vy = joyDelta.y; }

            const mag = Math.sqrt(vx * vx + vy * vy);
            if (mag > 0 && !joyActive) { vx /= mag; vy /= mag; }
            if (vx !== 0 || vy !== 0) localFacing = Math.atan2(vy, vx);

            // Ghost frozen — no movement during countdown
            if (me.isGhost && gameState.ghostTimer > 0) {
                localX = me.x; localY = me.y;
                return;
            }

            const spd = me.isGhost ? 130 : 220;

            // Client-side wall collision for both ghost and survivor
            if (map && rooms) {
                const moved = clientMoveCollide(localX, localY, vx, vy, spd, dt, me.isGhost ? GHOST_RADIUS_C : PLAYER_RADIUS, me.isGhost);
                localX = moved.x; localY = moved.y;
            } else {
                localX += vx * spd * dt; localY += vy * spd * dt;
            }

            localX = Math.max(PLAYER_RADIUS, Math.min(MAP_W * TILE_SIZE - PLAYER_RADIUS, localX));
            localY = Math.max(PLAYER_RADIUS, Math.min(MAP_H * TILE_SIZE - PLAYER_RADIUS, localY));

            // Write predicted position so draw loop uses it immediately
            me.x = localX; me.y = localY; me.vx = vx; me.vy = vy; me.facing = localFacing;
            localVx = vx; localVy = vy;

            // Check near bed (survivors only) — can enter unowned rooms freely now
            if (!me.isGhost && !hasSlepped && rooms) {
                let nearIdx = -1;
                for (let i = 0; i < rooms.length; i++) {
                    const r = rooms[i];
                    if (r.owner) continue;
                    // Check proximity to bed OR door tile
                    const bpx = r.bed.x * TILE_SIZE + TILE_SIZE / 2, bpy = r.bed.y * TILE_SIZE + TILE_SIZE / 2;
                    const dpx = r.door.x * TILE_SIZE + TILE_SIZE / 2, dpy = r.door.y * TILE_SIZE + TILE_SIZE / 2;
                    if (dist(localX, localY, bpx, bpy) < TILE_SIZE * 2 || dist(localX, localY, dpx, dpy) < TILE_SIZE * 1.2) {
                        nearIdx = i; break;
                    }
                }
                const btn = document.getElementById('sleep-btn');
                if (nearIdx >= 0) { btn.style.display = 'block'; btn.onclick = () => doSleep(nearIdx); }
                else btn.style.display = 'none';
            }
        }

        }

        // Shared handler for both click and touch-tap on canvas
        function handleCanvasTap(clientX, clientY) {
            const me = gameState?.players?.[socket.id];
            if (!me || me.state !== 'SLEEPING' || me.room === null) return;
            const r = rooms[me.room];
            if (!r) return;
            const dcx = Math.max(0, Math.min(camX - W / 2, MAP_W * TILE_SIZE - W));
            const dcy = Math.max(0, Math.min(camY - H / 2, MAP_H * TILE_SIZE - H));
            const wx = clientX + dcx, wy = clientY + dcy;
            const tx = Math.floor(wx / TILE_SIZE), ty = Math.floor(wy / TILE_SIZE);
            if ((tx >= r.x && tx < r.x + r.w && ty >= r.y && ty < r.y + r.h) || (tx === r.door.x && ty === r.door.y)) {
                const ti = { roomIdx: me.room, x: tx, y: ty, type: 'empty', obj: null };
                if (tx === r.bed.x && ty === r.bed.y) { ti.type = 'bed'; ti.obj = r.bed; }
                else if (tx === r.door.x && ty === r.door.y) { ti.type = 'door'; ti.obj = r.door; }
                else {
                    const tur = r.turrets.find(t => t.x === tx && t.y === ty);
                    const gen = r.generators.find(t => t.x === tx && t.y === ty);
                    const mine = r.mines.find(t => t.x === tx && t.y === ty);
                    if (tur) { ti.type = 'turret'; ti.obj = tur; }
                    else if (gen) { ti.type = 'generator'; ti.obj = gen; }
                    else if (mine) { ti.type = 'mine'; ti.obj = mine; }
                }
                openActionPanel(ti);
            } else {
                document.getElementById('action-panel').style.display = 'none'; selectedTile = null;
            }
        }

        // Canvas click → open build/upgrade (desktop)
        canvas.addEventListener('click', e => handleCanvasTap(e.clientX, e.clientY));

        // Canvas touch tap → open build/upgrade (mobile)
        let _tapStart = { x: 0, y: 0, t: 0 };
        canvas.addEventListener('touchstart', e => {
            const t = e.changedTouches[0];
            _tapStart = { x: t.clientX, y: t.clientY, t: Date.now() };
        }, { passive: true });
        canvas.addEventListener('touchend', e => {
            const t = e.changedTouches[0];
            const dx = t.clientX - _tapStart.x, dy = t.clientY - _tapStart.y;
            const moved = Math.sqrt(dx * dx + dy * dy);
            // Only fire if finger barely moved (tap, not swipe/pan) and quick
            if (moved < 12 && Date.now() - _tapStart.t < 300) {
                handleCanvasTap(t.clientX, t.clientY);
            }
        }, { passive: true });

            const scaleX = MM_W / MAP_W;
            const scaleY = MM_H / MAP_H;

            // Walls
            for (let y = 0; y < MAP_H; y++) for (let x = 0; x < MAP_W; x++) {
                if (map[y][x] === 1) {
                    mmCtx.fillStyle = '#1e293b';
                    mmCtx.fillRect(x * scaleX, y * scaleY, scaleX + 0.5, scaleY + 0.5);
                }
            }

            // Rooms
            for (const r of rooms) {
                mmCtx.fillStyle = r.owner ? 'rgba(180,100,20,0.4)' : 'rgba(100,60,20,0.25)';
                mmCtx.fillRect(r.x * scaleX, r.y * scaleY, r.w * scaleX, r.h * scaleY);
                // Door
                const dpct = r.door.hp / r.door.maxHp;
                mmCtx.fillStyle = dpct > 0.5 ? '#16a34a' : dpct > 0.25 ? '#f59e0b' : '#dc2626';
                mmCtx.fillRect(r.door.x * scaleX, r.door.y * scaleY, scaleX + 0.5, scaleY + 0.5);
            }

            // Camera viewport box
            const dcx = camX - W / 2, dcy = camY - H / 2;
            mmCtx.strokeStyle = 'rgba(255,255,255,0.2)';
            mmCtx.lineWidth = 1;
            mmCtx.strokeRect(
                Math.max(0, dcx / TILE_SIZE * scaleX),
                Math.max(0, dcy / TILE_SIZE * scaleY),
                Math.min(MM_W, W / TILE_SIZE * scaleX),
                Math.min(MM_H, H / TILE_SIZE * scaleY)
            );

            // Players
            for (const p of Object.values(gameState.players)) {
                if (p.state === 'DEAD') continue;
                const mx = p.x / TILE_SIZE * scaleX;
                const my = p.y / TILE_SIZE * scaleY;
                if (p.isGhost) {
                    const isFrozen = gameState.ghostTimer > 0;
                    mmCtx.fillStyle = isFrozen ? '#7dd3fc' : '#ef4444';
                    mmCtx.beginPath(); mmCtx.arc(mx, my, 3.5, 0, Math.PI * 2); mmCtx.fill();
                    // Red ring pulse
                    if (!isFrozen) {
                        const pulse = 0.4 + Math.abs(Math.sin(Date.now() / 300)) * 0.6;
                        mmCtx.strokeStyle = `rgba(239,68,68,${pulse})`; mmCtx.lineWidth = 1.5;
                        mmCtx.beginPath(); mmCtx.arc(mx, my, 5.5, 0, Math.PI * 2); mmCtx.stroke();
                    }
                } else {
                    const isMe = p.id === socket.id;
                    mmCtx.fillStyle = isMe ? '#ffffff' : (p.color || '#38bdf8');
                    mmCtx.beginPath(); mmCtx.arc(mx, my, isMe ? 3 : 2, 0, Math.PI * 2); mmCtx.fill();
                }
            }
        }

        // ── GHOST ABILITY ─────────────────────────────────────────────────
        window.ghostUseNet = function () {
            socket.emit('player_use_net', (res) => {
                if (!res?.ok) {
                    const btn = document.getElementById('ga-net');
                    btn.classList.add('disabled');
                }
            });
        };

        // ── GAME LOOP ─────────────────────────────────────────────────────
        let lastTime = 0, gameLoopRunning = false;
        function startGameLoop() {
            if (gameLoopRunning) return;
            gameLoopRunning = true;
            lastTime = performance.now();
            requestAnimationFrame(gameLoop);
        }
        function gameLoop(t) {
            if (!gameLoopRunning) return;
            const dt = Math.min((t - lastTime) / 1000, 0.1);
            lastTime = t;

            localMove(dt);
            updateCamera(dt);
            sendMyPosition(dt);

            // particles + floatingTexts age
            for (let i = particles.length - 1; i >= 0; i--) { let p = particles[i]; p.x += p.vx; p.y += p.vy; p.life -= dt; if (p.life <= 0) particles.splice(i, 1); }
            for (let i = floatingTexts.length - 1; i >= 0; i--) { floatingTexts[i].life -= dt; if (floatingTexts[i].life <= 0) floatingTexts.splice(i, 1); }

            draw();
            drawMinimap();
            requestAnimationFrame(gameLoop);
        }

        // ── DRAW ─────────────────────────────────────────────────────────
        function draw() {
            if (!map || !rooms || !gameState) return;
            // Camera is clamped to [W/2, mapW-W/2], so offset is camX - W/2
            const dcx = Math.max(0, Math.min(camX - W / 2, MAP_W * TILE_SIZE - W));
            const dcy = Math.max(0, Math.min(camY - H / 2, MAP_H * TILE_SIZE - H));

            ctx.fillStyle = '#060b14'; ctx.fillRect(0, 0, W, H);
            ctx.save(); ctx.translate(-dcx, -dcy);

            const sc = Math.floor(dcx / TILE_SIZE), ec = sc + Math.ceil(W / TILE_SIZE) + 2;
            const sr = Math.floor(dcy / TILE_SIZE), er = sr + Math.ceil(H / TILE_SIZE) + 2;

            // MAP
            for (let y = sr; y < er; y++) for (let x = sc; x < ec; x++) {
                if (y < 0 || y >= MAP_H || x < 0 || x >= MAP_W) continue;
                const t = map[y][x], px = x * TILE_SIZE, py = y * TILE_SIZE;
                if (t === 1) {
                    ctx.fillStyle = '#0f172a'; ctx.fillRect(px, py, TILE_SIZE, TILE_SIZE);
                    ctx.fillStyle = '#1e293b';
                    const bW = TILE_SIZE / 2 - 2, bH = TILE_SIZE / 4 - 2;
                    ctx.fillRect(px + 2, py + 2, bW, bH); ctx.fillRect(px + TILE_SIZE / 2 + 1, py + 2, bW, bH);
                    ctx.fillRect(px + 2, py + TILE_SIZE / 4 + 1, TILE_SIZE - 4, bH);
                    ctx.fillRect(px + 2, py + TILE_SIZE / 2 + 1, bW, bH); ctx.fillRect(px + TILE_SIZE / 2 + 1, py + TILE_SIZE / 2 + 1, bW, bH);
                } else {
                    ctx.fillStyle = ((x + y) % 2 === 0) ? '#0a0f1e' : '#0f1628'; ctx.fillRect(px, py, TILE_SIZE, TILE_SIZE);
                }
            }

            // ROOMS
            for (let ri = 0; ri < rooms.length; ri++) {
                const r = rooms[ri];
                const rx = r.x * TILE_SIZE, ry = r.y * TILE_SIZE;
                if (rx + r.w * TILE_SIZE < dcx || rx > dcx + W || ry + r.h * TILE_SIZE < dcy || ry > dcy + H) continue;

                ctx.fillStyle = r.owner ? '#1a0f05' : '#0f0a05'; ctx.fillRect(rx, ry, r.w * TILE_SIZE, r.h * TILE_SIZE);

                // Danger aura: pulse red when ghost is attacking or inside this room
                const ghostPlayer = gameState && Object.values(gameState.players).find(p => p.isGhost);
                const ghostTargetRoom = ghostPlayer && gameState.players[ghostPlayer.ghostTarget]?.room;
                const isUnderAttack = ghostPlayer && ghostTargetRoom === ri &&
                    (ghostPlayer.ghostState === 'ATTACKING' || ghostPlayer.ghostState === 'INSIDE' || ghostPlayer.ghostState === 'CROSSING');
                if (isUnderAttack) {
                    const danger = 0.15 + Math.abs(Math.sin(Date.now() / 150)) * 0.25;
                    ctx.fillStyle = `rgba(239,68,68,${danger})`; ctx.fillRect(rx, ry, r.w * TILE_SIZE, r.h * TILE_SIZE);
                    ctx.strokeStyle = `rgba(239,68,68,${danger * 2})`; ctx.lineWidth = 3;
                    ctx.strokeRect(rx, ry, r.w * TILE_SIZE, r.h * TILE_SIZE);
                }
                ctx.strokeStyle = r.owner ? 'rgba(180,100,20,0.35)' : 'rgba(100,60,20,0.25)'; ctx.lineWidth = 1;
                for (let i = 0; i <= r.w; i++) { ctx.beginPath(); ctx.moveTo(rx + i * TILE_SIZE, ry); ctx.lineTo(rx + i * TILE_SIZE, ry + r.h * TILE_SIZE); ctx.stroke(); }
                for (let i = 0; i <= r.h; i++) { ctx.beginPath(); ctx.moveTo(rx, ry + i * TILE_SIZE); ctx.lineTo(rx + r.w * TILE_SIZE, ry + i * TILE_SIZE); ctx.stroke(); }

                // Selected highlight
                if (selectedTile && selectedTile.roomIdx === ri) {
                    ctx.fillStyle = 'rgba(255,255,255,0.1)'; ctx.fillRect(selectedTile.x * TILE_SIZE, selectedTile.y * TILE_SIZE, TILE_SIZE, TILE_SIZE);
                    ctx.strokeStyle = 'rgba(255,255,255,0.8)'; ctx.lineWidth = 2; ctx.strokeRect(selectedTile.x * TILE_SIZE + 1, selectedTile.y * TILE_SIZE + 1, TILE_SIZE - 2, TILE_SIZE - 2);
                }

                // BED
                const bx = r.bed.x * TILE_SIZE, by = r.bed.y * TILE_SIZE;
                ctx.fillStyle = MAT_COLORS[r.bed.level] || '#78350f'; ctx.fillRect(bx + 2, by + 2, TILE_SIZE - 4, TILE_SIZE - 4);
                ctx.fillStyle = '#f8fafc'; ctx.fillRect(bx + 4, by + 4, TILE_SIZE - 8, TILE_SIZE - 8);
                const bColor = r.owner ? (TIER_COLORS[r.bed.level] || '#3b82f6') : '#475569';
                ctx.fillStyle = bColor; ctx.fillRect(bx + 4, by + TILE_SIZE / 3, TILE_SIZE - 8, TILE_SIZE * 2 / 3 - 4);
                ctx.fillStyle = '#e2e8f0'; ctx.fillRect(bx + 8, by + 6, TILE_SIZE - 16, 12);

                // MINES
                for (const m of r.mines) {
                    const mx = m.x * TILE_SIZE, my = m.y * TILE_SIZE;
                    const mCol = { copper: '#b45309', silver: '#94a3b8', gold: '#eab308' }[m.type] || '#78350f';
                    ctx.fillStyle = '#1c1917'; ctx.fillRect(mx + 3, my + 3, TILE_SIZE - 6, TILE_SIZE - 6);
                    ctx.fillStyle = mCol; ctx.fillRect(mx + 6, my + 6, TILE_SIZE - 12, TILE_SIZE - 12);
                    const pulse = 0.5 + Math.abs(Math.sin(Date.now() / 400 + m.x)) * 0.5;
                    ctx.fillStyle = `rgba(255,255,255,${pulse * 0.25})`; ctx.beginPath(); ctx.arc(mx + TILE_SIZE / 2, my + TILE_SIZE / 2, 7, 0, Math.PI * 2); ctx.fill();
                    ctx.fillStyle = '#fff'; ctx.font = '10px Arial'; ctx.textAlign = 'center';
                    ctx.fillText({ copper: '⛏️', silver: '🪨', gold: '✨' }[m.type] || '?', mx + TILE_SIZE / 2, my + TILE_SIZE / 2 + 4);
                    ctx.font = 'bold 9px Arial'; ctx.fillText(`Lv${m.level}`, mx + TILE_SIZE / 2, my + TILE_SIZE - 5);
                }

                // GENERATORS
                for (const g of r.generators) {
                    const cx = g.x * TILE_SIZE + TILE_SIZE / 2, cy = g.y * TILE_SIZE + TILE_SIZE / 2;
                    ctx.fillStyle = MAT_COLORS[g.level] || '#475569'; ctx.fillRect(g.x * TILE_SIZE + 5, g.y * TILE_SIZE + 5, TILE_SIZE - 10, TILE_SIZE - 10);
                    const pulse = Math.abs(Math.sin(Date.now() / 300));
                    ctx.fillStyle = `rgba(34,211,238,${.4 + pulse * .6})`; ctx.beginPath(); ctx.arc(cx, cy, 8, 0, Math.PI * 2); ctx.fill();
                    ctx.fillStyle = '#fff'; ctx.font = 'bold 10px Arial'; ctx.textAlign = 'center'; ctx.fillText(`Lv${g.level}`, cx, cy - 12);
                }

                // TURRETS
                for (const t of r.turrets) {
                    const cx = t.x * TILE_SIZE + TILE_SIZE / 2, cy = t.y * TILE_SIZE + TILE_SIZE / 2;
                    const tColor = TIER_COLORS[t.level] || '#ef4444';
                    const br = 10 + Math.min(t.level, 12) * 0.6;
                    ctx.fillStyle = MAT_COLORS[t.level] || '#334155'; ctx.beginPath(); ctx.arc(cx, cy, br, 0, Math.PI * 2); ctx.fill();
                    ctx.strokeStyle = tColor; ctx.lineWidth = 2; ctx.stroke();
                    ctx.save(); ctx.translate(cx, cy); ctx.rotate(t.angle || 0);
                    const bl = 12 + t.level * 1.5, bw = 4 + Math.floor(t.level / 3) * 1.5;
                    ctx.fillStyle = '#cbd5e1'; ctx.fillRect(2, -bw / 2, bl, bw);
                    ctx.fillStyle = tColor; ctx.beginPath(); ctx.arc(0, 0, br * .7, 0, Math.PI * 2); ctx.fill();
                    ctx.restore();
                    ctx.fillStyle = '#fff'; ctx.font = 'bold 9px Arial'; ctx.textAlign = 'center'; ctx.fillText(`Lv${t.level}`, cx, cy - br - 5);
                }

                // Tool indicators
                let tiY = ry - 18;
                if (r._hasGrenadeLauncher) { ctx.font = '12px Arial'; ctx.textAlign = 'left'; ctx.fillText('💣', rx + 3, tiY + 12); tiY += 18; }
                if (r._hasGhostNet) { ctx.fillText('🕸️', rx + 3, tiY + 12); tiY += 18; }
                if (r._hasShield) { ctx.fillText('🛡️', rx + 3, tiY + 12); }

                // DOOR
                const ddx = r.door.x * TILE_SIZE, ddy = r.door.y * TILE_SIZE;
                if (r.door.hp > 0) {
                    let dColor = r.owner ? (r.door.level > 0 ? (MAT_COLORS[r.door.level] || '#ca8a04') : '#92400e') : '#d97706';
                    // Flash white on hit
                    if (r.door.hitTimer > 0) {
                        const flash = r.door.hitTimer / 0.2;
                        ctx.fillStyle = dColor; ctx.fillRect(ddx, ddy, TILE_SIZE, TILE_SIZE);
                        ctx.fillStyle = `rgba(255,255,255,${flash * 0.9})`; ctx.fillRect(ddx, ddy, TILE_SIZE, TILE_SIZE);
                        // Crack sparks when being attacked
                        for (let sp = 0; sp < 4; sp++) {
                            const sx = ddx + 4 + Math.random() * (TILE_SIZE - 8);
                            const sy = ddy + 4 + Math.random() * (TILE_SIZE - 8);
                            ctx.fillStyle = '#fbbf24'; ctx.fillRect(sx, sy, 3, 3);
                        }
                    } else {
                        ctx.fillStyle = dColor; ctx.fillRect(ddx, ddy, TILE_SIZE, TILE_SIZE);
                    }
                    // Door planks texture
                    if (r.owner) {
                        ctx.strokeStyle = 'rgba(0,0,0,0.25)'; ctx.lineWidth = 1.5;
                        for (let pi = 1; pi < 3; pi++) {
                            ctx.beginPath(); ctx.moveTo(ddx, ddy + pi * (TILE_SIZE / 3)); ctx.lineTo(ddx + TILE_SIZE, ddy + pi * (TILE_SIZE / 3)); ctx.stroke();
                        }
                        ctx.beginPath(); ctx.moveTo(ddx + TILE_SIZE / 2, ddy); ctx.lineTo(ddx + TILE_SIZE / 2, ddy + TILE_SIZE); ctx.stroke();
                    }
                    // Shield pulse
                    if (r.door.shieldActive || r._hasShield) {
                        ctx.strokeStyle = `rgba(96,165,250,${0.5 + Math.sin(Date.now() / 200) * 0.5})`; ctx.lineWidth = 3;
                        ctx.strokeRect(ddx + 2, ddy + 2, TILE_SIZE - 4, TILE_SIZE - 4);
                        ctx.fillStyle = `rgba(96,165,250,0.08)`; ctx.fillRect(ddx, ddy, TILE_SIZE, TILE_SIZE);
                    }
                    // HP bar above door
                    if (r.owner) {
                        ctx.fillStyle = 'rgba(0,0,0,0.8)'; ctx.fillRect(ddx, ddy - 12, TILE_SIZE, 8);
                        const hpPct = r.door.hp / r.door.maxHp;
                        ctx.fillStyle = hpPct > 0.5 ? '#16a34a' : hpPct > 0.25 ? '#f59e0b' : '#dc2626';
                        ctx.fillRect(ddx + 1, ddy - 11, (TILE_SIZE - 2) * hpPct, 6);
                    }
                } else {
                    // Broken door — rubble effect
                    ctx.fillStyle = 'rgba(0,0,0,0.6)'; ctx.fillRect(ddx, ddy, TILE_SIZE, TILE_SIZE);
                    // Rubble chunks
                    const rubbleSeeds = [[8, 6, 12, 8], [22, 10, 10, 7], [6, 22, 14, 6], [26, 18, 10, 9], [14, 28, 10, 8], [4, 34, 8, 6]];
                    for (const [rx2, ry2, rw2, rh2] of rubbleSeeds) {
                        ctx.fillStyle = '#44403c'; ctx.fillRect(ddx + rx2, ddy + ry2, rw2, rh2);
                        ctx.fillStyle = '#57534e'; ctx.fillRect(ddx + rx2 + 1, ddy + ry2 + 1, rw2 - 2, 2);
                    }
                    // Red glow on broken door = ghost can enter
                    const brokenPulse = 0.3 + Math.abs(Math.sin(Date.now() / 400)) * 0.3;
                    ctx.fillStyle = `rgba(239,68,68,${brokenPulse})`; ctx.fillRect(ddx, ddy, TILE_SIZE, TILE_SIZE);
                }
            }

            // PLAYERS
            for (const p of Object.values(gameState.players)) {
                if (p.state === 'DEAD') continue;
                drawEntity(p);
            }

            // PARTICLES
            for (const p of particles) {
                ctx.fillStyle = p.color; ctx.globalAlpha = p.life / p.maxLife;
                ctx.beginPath(); ctx.arc(p.x, p.y, 4, 0, Math.PI * 2); ctx.fill();
            }
            ctx.globalAlpha = 1;

            // FLOATING TEXTS
            for (const ft of floatingTexts) {
                ctx.globalAlpha = ft.life / ft.maxLife;
                ctx.font = 'bold 15px Orbitron,Arial'; ctx.textAlign = 'center';
                ctx.shadowColor = 'rgba(0,0,0,0.9)'; ctx.shadowBlur = 6; ctx.lineWidth = 3;
                ctx.strokeStyle = 'rgba(0,0,0,0.8)'; ctx.strokeText(ft.text, ft.x, ft.y - (1 - ft.life / ft.maxLife) * 30);
                ctx.fillStyle = ft.color; ctx.fillText(ft.text, ft.x, ft.y - (1 - ft.life / ft.maxLife) * 30);
                ctx.shadowBlur = 0;
            }
            ctx.globalAlpha = 1;
            ctx.restore(); // ← end world-space transform

            // Ghost-player overlay: red vignette + state label (screen space)
            const myData = gameState?.players?.[socket.id];
            if (myData?.isGhost && myData.state !== 'DEAD') {
                // Red vignette edges
                const vign = ctx.createRadialGradient(W / 2, H / 2, H * 0.3, W / 2, H / 2, H * 0.85);
                vign.addColorStop(0, 'rgba(0,0,0,0)');
                vign.addColorStop(1, 'rgba(180,0,0,0.35)');
                ctx.fillStyle = vign; ctx.fillRect(0, 0, W, H);
                // Ghost state badge
                const stateLabels = { WAITING: '❄ FROZEN', HUNTING: '👁 HUNTING', APPROACHING_DOOR: '🏃 APPROACHING', ATTACKING: '💥 ATTACKING DOOR', CROSSING: '🌀 ENTERING', INSIDE: '🩸 INSIDE' };
                const stateColors = { WAITING: '#7dd3fc', HUNTING: '#fbbf24', APPROACHING_DOOR: '#fb923c', ATTACKING: '#ef4444', CROSSING: '#a855f7', INSIDE: '#dc2626' };
                const sLabel = stateLabels[myData.ghostState] || myData.ghostState;
                const sColor = stateColors[myData.ghostState] || '#fff';
                ctx.font = 'bold 13px Orbitron,monospace'; ctx.textAlign = 'center';
                const sw = ctx.measureText(sLabel).width + 24;
                ctx.fillStyle = 'rgba(0,0,0,0.75)'; ctx.beginPath();
                ctx.roundRect(W / 2 - sw / 2, H - 54, sw, 28, 6); ctx.fill();
                ctx.fillStyle = sColor; ctx.fillText(sLabel, W / 2, H - 35);
                if (myData.ghostStunTimer > 0) {
                    ctx.fillStyle = 'rgba(100,200,255,0.25)'; ctx.fillRect(0, 0, W, H);
                    ctx.font = 'bold 18px Orbitron,monospace'; ctx.fillStyle = '#7dd3fc'; ctx.textAlign = 'center';
                    ctx.fillText(`🕸️ STUNNED ${myData.ghostStunTimer.toFixed(1)}s`, W / 2, H / 2 - 20);
                }
            }
        }  // end draw()

        function drawEntity(p) {
            const isMe = p.id === socket.id;
            const isFrozen = p.isGhost && gameState && gameState.ghostTimer > 0;

            ctx.save(); ctx.translate(p.x, p.y);

            if (p.isGhost) {
                // Stun ring (ghost net)
                if (p.ghostStunTimer > 0) {
                    ctx.strokeStyle = `rgba(100,200,255,0.7)`; ctx.lineWidth = 3;
                    ctx.beginPath(); ctx.arc(0, 0, 28, 0, Math.PI * 2); ctx.stroke();
                }

                // Frozen cage effect while waiting
                if (isFrozen) {
                    ctx.globalAlpha = 0.5;
                    // Pulsing ice cage
                    const pulse = 0.5 + Math.abs(Math.sin(Date.now() / 300)) * 0.5;
                    ctx.strokeStyle = `rgba(120,200,255,${0.4 + pulse * 0.5})`; ctx.lineWidth = 2;
                    for (let i = 0; i < 4; i++) {
                        ctx.beginPath();
                        ctx.arc(0, 0, 22 + i * 6, 0, Math.PI * 2);
                        ctx.stroke();
                    }
                    // Countdown text above
                    ctx.globalAlpha = 1;
                    ctx.font = 'bold 11px Orbitron,monospace'; ctx.textAlign = 'center';
                    ctx.fillStyle = '#7dd3fc';
                    ctx.fillText(`❄ ${Math.ceil(gameState.ghostTimer)}s`, 0, -38);
                }

                // Ghost glow
                const glow = ctx.createRadialGradient(0, 0, 10, 0, 0, 40);
                glow.addColorStop(0, isFrozen ? 'rgba(100,180,255,0.4)' : 'rgba(239,68,68,0.6)');
                glow.addColorStop(1, 'rgba(0,0,0,0)');
                ctx.fillStyle = glow; ctx.beginPath(); ctx.arc(0, 0, 40, 0, Math.PI * 2); ctx.fill();

                // Ghost body
                ctx.fillStyle = isFrozen ? '#1e40af' : '#991b1b';
                ctx.beginPath(); ctx.arc(0, -10, 20, Math.PI, 0);
                ctx.lineTo(20, 22);
                for (let i = 1; i <= 4; i++) { const wx = 20 - (i * 10), wy = 22 + Math.sin((Date.now() / 200) + i) * 5; ctx.lineTo(wx, wy); }
                ctx.lineTo(-20, 22); ctx.closePath(); ctx.fill();

                // Eyes
                ctx.fillStyle = isFrozen ? '#93c5fd' : '#fde047';
                ctx.beginPath(); ctx.ellipse(-6, -12, 4, 2, 0, 0, Math.PI * 2); ctx.fill();
                ctx.beginPath(); ctx.ellipse(6, -12, 4, 2, 0, 0, Math.PI * 2); ctx.fill();

                ctx.globalAlpha = 1;
                ctx.restore();

                // HP bar (only show when not frozen)
                if (!isFrozen) {
                    ctx.fillStyle = 'rgba(0,0,0,0.8)'; ctx.fillRect(p.x - 24, p.y - 50, 48, 8);
                    ctx.fillStyle = '#ef4444'; ctx.fillRect(p.x - 23, p.y - 49, 46 * (Math.max(0, p.hp) / p.maxHp), 6);
                }
                ctx.fillStyle = isFrozen ? '#7dd3fc' : '#fca5a5';
                ctx.font = 'bold 10px Orbitron,monospace'; ctx.textAlign = 'center';
                ctx.fillText(`👻 ${p.name}`, p.x, p.y - 54);
                return;
            }

            if (p.state === 'SLEEPING') {
                ctx.fillStyle = '#fca5a5'; ctx.beginPath(); ctx.arc(0, -8, 7, 0, Math.PI * 2); ctx.fill();
                ctx.fillStyle = '#fff'; ctx.font = 'bold 14px Arial'; ctx.textAlign = 'center'; ctx.fillText('Z', 8, -15);
            } else {
                ctx.rotate(p.facing || 0);
                ctx.fillStyle = p.color || '#38bdf8'; ctx.beginPath(); ctx.arc(-2, 0, 10, 0, Math.PI * 2); ctx.fill(); ctx.beginPath(); ctx.arc(2, 0, 10, 0, Math.PI * 2); ctx.fill();
                ctx.fillStyle = '#fca5a5'; ctx.beginPath(); ctx.arc(4, 0, 8, 0, Math.PI * 2); ctx.fill();
                ctx.fillStyle = isMe ? '#0f172a' : '#78350f'; ctx.beginPath(); ctx.arc(3, 0, 8.5, Math.PI / 2, Math.PI * 1.5); ctx.fill();
                ctx.fillStyle = '#0f172a'; ctx.beginPath(); ctx.arc(8, -3, 1.5, 0, Math.PI * 2); ctx.fill(); ctx.beginPath(); ctx.arc(8, 3, 1.5, 0, Math.PI * 2); ctx.fill();
            }
            ctx.restore();

            // Name tag
            ctx.fillStyle = 'rgba(0,0,0,0.7)'; ctx.fillRect(p.x - 22, p.y - 32, 44, 14);
            ctx.font = 'bold 9px Orbitron,monospace'; ctx.textAlign = 'center';
            ctx.fillStyle = isMe ? '#38bdf8' : 'rgba(200,210,240,0.9)';
            ctx.fillText(p.name.length > 7 ? p.name.slice(0, 7) + '…' : p.name, p.x, p.y - 22);
        }
